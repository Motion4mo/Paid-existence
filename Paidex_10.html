    let currentCode = '';
    const validCodes = ['333', '1337', '2025', '0000'];
    
    function addToCode(digit) {
        if (currentCode.length < 4) {
            currentCode += digit;
            document.getElementById('accessCode').value = currentCode;
        }
    }
    
    function clearCode() {
        currentCode = '';
        document.getElementById('accessCode').value = '';
    }
    
    function checkAccess() {
        const code = document.getElementById('accessCode').value;
        if (validCodes.includes(code)) {
            showPage('hub');
            // Add access granted effect
            document.body.style.animation = 'glitch 0.5s ease-in-out';
            setTimeout(() => {
                document.body.style.animation = '';
            }, 500);
        } else {
            // Add denied effect
            document.getElementById('accessCode').style.borderColor = '#ff0000';
            document.getElementById('accessCode').style.boxShadow = '0 0 20px rgba(255, 0, 0, 0.5)';
            setTimeout(() => {
                document.getElementById('accessCode').style.borderColor = '#8B0000';
                document.getElementById('accessCode').style.boxShadow = 'none';
            }, 1000);
            clearCode();
        }
    }
    
    function showPage(pageId) {
        // Hide all pages
        const pages = document.querySelectorAll('.page');
        pages.forEach(page => page.classList.remove('active'));
        
        // Show selected page
        document.getElementById(pageId).classList.add('active');
        
        // Reset scroll position
        window.scrollTo(0, 0);
    }
    
    // Allow Enter key to submit access code
    document.getElementById('accessCode').addEventListener('keypress', function(e) {
        if (e.key === 'Enter') {
            checkAccess();
        }
    });
    
    // Add some interactive effects
    document.addEventListener('DOMContentLoaded', function() {
        // Add random glitch effects
        setInterval(() => {
            const glitchElements = document.querySelectorAll('.glitch');

                    setTimeout(() => {

                }
            });
